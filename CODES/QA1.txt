TABLE CONTAINS 
EMP ID 
CREATE FIRST NAME AND LAST NAME
NEW SALARY AND OLD SALARY
NEW SALARY=IF JOBID CONTAINS MGR GIVE 10% HIKE
IF THE JOBID CONTAINS ACCOUNT GIVE 15% HIKE
IF THE JOBID CONTAINS CLERK GIVE 20% HIKE 
OTHERS GIVE 5% HIKE





---------------------------------------------------------------------
CREATE OR REPLACE PROCEDURE CAP_PROC(p_EMPID VARCHAR2)
AS
 	V_EMPID CAP.EMPID%TYPE;
 	V_FIRSTNAME CAP.FIRSTNAME%TYPE;
 	V_SECONDNAME CAP.SECONDNAME%TYPE;
 	V_SALARY CAP.SALARY%TYPE;
	V_NEWSALARY CAP.SALARY%TYPE;
	V_JOBID CAP.JOBID%TYPE;

 BEGIN
	 SELECT EMPID,FIRSTNAME,SECONDNAME,JOBID,SALARY INTO V_EMPID,V_FIRSTNAME,V_SECONDNAME,V_JOBID,V_SALARY FROM CAP WHERE EMPID=p_EMPID;
 	DBMS_OUTPUT.PUT_LINE('EMP ID IS:'||V_EMPID);
 	DBMS_OUTPUT.PUT_LINE('FIRST NAME IS:'||V_FIRSTNAME);
 	DBMS_OUTPUT.PUT_LINE('SECONDNAME IS :'||V_SECONDNAME);
 	DBMS_OUTPUT.PUT_LINE('OLD SALARY  IS :'||V_SALARY);
IF V_JOBID='MGR' THEN
	DBMS_OUTPUT.PUT_LINE('NEW SALARY IS :'||1.1*V_SALARY);
	UPDATE CAP SET SALARY=SALARY*1.1 WHERE EMPID=P_EMPID;
ELSIF V_JOBID='CLERK'THEN
	DBMS_OUTPUT.PUT_LINE('NEW SALARY IS :'||1.15*V_SALARY);
	UPDATE CAP SET SALARY=SALARY*1.15 WHERE EMPID=P_EMPID;
ELSIF V_JOBID='ACCOUNT' THEN
	DBMS_OUTPUT.PUT_LINE('NEW SALARY IS :'||1.2*V_SALARY);
	UPDATE CAP SET SALARY=SALARY*1.2 WHERE EMPID=P_EMPID;
ELSE
	DBMS_OUTPUT.PUT_LINE('NEW SALARY IS :'||1.05*V_SALARY);
	UPDATE CAP SET SALARY=SALARY*1.05 WHERE EMPID=P_EMPID;
END IF;
EXCEPTION

	WHEN OTHERS THEN
	DBMS_OUTPUT.PUT_LINE('SOMETHING WENT WRONG');
END;
